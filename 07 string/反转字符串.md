### 3.1 C++

#### 双指针 + 迭代

```c++
// version 1.0
class Solution {
public:
    string reverseString(string s) {
        
        int i, j;
        int str_len = s.size();
        
        char tmp;
        
        for(i=0; i<str_len/2; i++){
            tmp = s[i];
            s[i] = s[str_len-i-1];
            s[str_len-i-1] = tmp;
        }
        
        return s;  
    }
};
```

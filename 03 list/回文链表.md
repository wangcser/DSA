```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        int XOR = 0;
        int posi = 0;
        int list_len = 0;
        ListNode* cur = head;
        
        if(head == NULL) return true;
        
        while(head != NULL){
            XOR ^= head->val;
            list_len++;
            head = head->next;
        }
        
        if((list_len % 2) != 0){
            while(posi<=(list_len/2)-1){
                cur = cur->next;
                posi++;
            }
        }

        if((list_len % 2) == 0 && XOR == 0) return true;
        else if((list_len % 2) != 0 && cur->val == XOR) return true;
        else return false;
    }
};
```

```c++
/**

 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}

 */
class Solution {
public:
    bool isPalindrome(ListNode* head) {
        vector<int> check_list;
        int i = 0;
        ListNode* r_head = head;
        
        if(head == NULL) return true;
        
        while(head != NULL){
            check_list.push_back(head->val);
            head = head->next;
        }
        
        r_head = reverseList(r_head);
        //cout<<r_head->val<<endl;
        
        while(r_head != NULL){
            if(check_list[i] != r_head->val) return false;
            else{
                i++;
                r_head = r_head->next;
            }
        }
        
        return true;

    }
    
    ListNode* reverseList(ListNode* head) {
        
        if(head == NULL || head->next == NULL) return head;
        
        ListNode* rHead = reverseList(head->next);
        head->next->next = head;
        head->next = NULL;

        return rHead;
    }
};
```

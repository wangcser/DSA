# 数值的整数次方
> Author ID.9276

## 01 题目

实现 [pow(*x*, *n*)](https://www.cplusplus.com/reference/valarray/pow/) ，即计算 x 的 n 次幂函数。

**示例 1:**

```
输入: 2.00000, 10
输出: 1024.00000
```

**示例 2:**

```
输入: 2.10000, 3
输出: 9.26100
```

**示例 3:**

```
输入: 2.00000, -2
输出: 0.25000
解释: 2-2 = 1/22 = 1/4 = 0.25
```

**说明:**

- -100.0 < *x* < 100.0
- *n* 是 32 位有符号整数，其数值范围是 [−231, 231 − 1] 。

## 02 分析

class Solution {
public:
    double myPow(double x, int n) {
        
        if(isEqual(x, 0.0) && n < 0)
            return 0.0;
        
        if(isEqual(x, 1.0))
            return 1.0;
        
        double result;
        
        unsigned int uExp;
        
        if(n == -2147483648)
            uExp = 2147483648;
        else if(n < 0)
            uExp = (unsigned int) -n;
        else
            uExp = (unsigned int) n;
        
        result = powWithUnsignedExp(x, uExp);
        
        if(n < 0)
            return 1.0 / result;
        else
            return result;
    }
    
    
    double powWithUnsignedExp(double base, unsigned int exp) {
        double ans = 1.0;
        for(int i = 1; i <= exp; i++) {
            ans *= base;
            if(isEqual(ans, 0.0)) return 0.0;
            if(isEqual(ans, 1.0)) return 1.0;
        }
        
        return ans;
    }
    
    
    bool isEqual(double n1, double n2) {
        double eps = 0.0000001;
        if((n1 - n2) > - eps && (n1 - n2) < eps)
            return true;
        else 
            return false;
    }
};

class Solution {
public:
    double myPow(double x, int n) {
        
        if(isEqual(x, 0.0) && n < 0)
            return 0.0;
        
        if(isEqual(x, 1.0))
            return 1.0;
        
        double result;
        
        unsigned int uExp;
        
        if(n == -2147483648)
            uExp = 2147483648;
        else if(n < 0)
            uExp = (unsigned int) -n;
        else
            uExp = (unsigned int) n;
        
        result = powWithUnsignedExp(x, uExp);
        
        if(n < 0)
            return 1.0 / result;
        else
            return result;
    }
    
    
    double powWithUnsignedExp(double base, unsigned int exp) {
        
        if(exp == 0) return 1.0;
        if(exp == 1) return base;
        
        double ans = powWithUnsignedExp(base, exp >> 1);
        ans *= ans;
        
        if(exp & 0x1 == 1) // exp is odd
            ans *= base;
        
        return ans;
    }
    
    
    bool isEqual(double n1, double n2) {
        double eps = 0.0000001;
        if((n1 - n2) > - eps && (n1 - n2) < eps)
            return true;
        else 
            return false;
    }
};

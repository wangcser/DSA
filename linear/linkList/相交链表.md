```c++
//version 1.0 0.0%

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        
        // 1. detect loop: a.tail->next = b,
        // if there is loop, will detected
        // if not, will go to NULL
        ListNode* noLoop = NULL;
        if(headA == NULL || headB == NULL) return noLoop;
        
        ListNode* p = headA;
        while(p->next != NULL) {
            p = p->next;
        }
        cout<<p->val<<endl;
        // a.tail, create a loop in a and b.
        p->next = headB;
        cout<<p->next->val<<endl;
        

        ListNode* fast = headA;
        ListNode* slow = headA;
        
        do {
            if(fast == NULL) {
                p->next = NULL;
                return noLoop;
            }
            
            slow = slow->next;
            fast = fast->next;
            // this fast is the next of prev fast pointer.
            if(fast != NULL) fast = fast->next;
            else {
                p->next = NULL;
                return noLoop;
            }
            
        } while(fast != slow);
        
        fast = headA;
        while(fast != slow) {
            fast = fast->next;
            slow = slow->next;
        }
            
        p->next = NULL;
        
        return fast;
        
        
        
        
    }
};
```

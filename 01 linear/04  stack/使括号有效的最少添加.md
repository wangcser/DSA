## 04 核心代码

```c++
class Solution {
public:
    int minAddToMakeValid(string S) {
        stack<char> st;
        
        for(auto c : S) {
            if(!st.empty() && st.top() == '(' && c == ')')
                st.pop();
            else
                st.push(c);
        }
        
        return st.size();
    }
};
```

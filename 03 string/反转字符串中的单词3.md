## 01 题目

给定一个字符串，你需要反转字符串中每个单词的字符顺序，同时仍保留空格和单词的初始顺序。

**示例 1:**

```
输入: "Let's take LeetCode contest"
输出: "s'teL ekat edoCteeL tsetnoc" 
```

**注意：**在字符串中，每个单词由单个空格分隔，并且字符串中不会有任何额外的空格。

### 3.1 C++

```c++
//version 1.0 0.0%

class Solution {
public:
    string reverseWords(string s) {
        
        if(s.empty()) return s;
        
        // 1. remove space mark.
        removeSpace(s);
        
        // 2. inverse the string 
        inverse(s, 0, s.size()-1);
        
        // 3. reverse the word
        int i = 0;
        int j = i + 1;
        for( ; j < s.size(); j++){
            if(s[i] != ' ' && s[j] == ' ') {
                inverse(s, i, j-1);
                i = j;
            } else if(s[i] == ' ' && s[j] != ' ') i = j;   
        }
        
        // deal the last word.
        if(j == s.size()) inverse(s, i, j-1);
        
        return s;
    }
    
    void removeSpace(string& s) {
         // add front and end mark to remove front and end spaces.
        s = "a " + s + " a";
        
        // 1. remove the space, first and list char is 'a'
        int i = 0;
        bool flag = 0; // space flag
        
        for(int j = 0; j < s.size(); ) {
            if(s[j] != ' ') {
                flag = 1;
                s[i++] = s[j++];
            } else {
                if(flag == 1) {
                    s[i++] = ' ';
                    flag = 0;
                }
                j++;
            }
        }
        
        // clean the string.
        s.erase(i, s.size() - i);
        
        // debug the "    " case.
        if(s == "a a") s = "";
        else {
            // remove front and end mark.
            s.erase(0, 2);
            s.erase(s.size()-2, 2);
        }
    }
    
    void inverse(string& s, int i,int j) {
        
        char tmp;
        
        while(i < j) {
            tmp = s[i];
            s[i] = s[j];
            s[j] = tmp;
            i++;
            j--;
        }
    }
};
```

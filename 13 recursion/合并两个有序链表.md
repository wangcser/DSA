#   合并两个有序链表 

> Author：ID.9276

## 01 题目

将两个有序链表合并为一个新的有序链表并返回。新链表是通过拼接给定的两个链表的所有节点组成的。 

**示例：**

```
输入：1->2->4, 1->3->4
输出：1->1->2->3->4->4
```

## 02 分析

思路1：将 l1的节点插入到l2中

思路2：采用归并的思想

思路3：也可以迭代或者递归

## 03 题解

### 3.1 C++

```c++
// version 1.0
// 归并思路
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        
        ListNode* cur1 = l1;
        ListNode* cur2 = l2;
        ListNode* mergeList = NULL;
        ListNode* mH = NULL;
        ListNode* tmp;
        
        if(cur1->val < cur2->val){
                tmp = cur1->next;
                mergeList = cur1;
                mH = mergeList;
                cur1 = tmp;
                mergeList->next = NULL;
                mergeList = mergeList->next;
            }else{
                tmp = cur2->next;
                mergeList = cur2;
                mH = mergeList;
                cur2 = tmp;
                mergeList->next = NULL;
                mergeList = mergeList->next;
            
            }
        
        while(cur1 != NULL && cur2 != NULL){
            if(cur1->val < cur2->val){
                tmp = cur1->next;
                mergeList = cur1;
                cur1 = tmp;
                mergeList->next = NULL;
                mergeList = mergeList->next;
            }else{
                tmp = cur2->next;
                mergeList = cur2;
                cur2 = tmp;
                mergeList->next = NULL;
                mergeList = mergeList->next;
            }
        }
        
        return mH;
    }
};
```

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        
        if(l1 == NULL) return l2;
        if(l2 == NULL) return l1;
        
        ListNode* mergeHead = NULL;
        
        if(l1->val < l2->val){
            mergeHead = l1;
            mergeHead->next = mergeTwoLists(l1->next, l2);
        }else{
            mergeHead = l2;
            mergeHead->next = mergeTwoLists(l1, l2->next);
        }
        
        return mergeHead;
    }
};
```
